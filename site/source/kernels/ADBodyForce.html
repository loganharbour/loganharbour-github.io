    ny = 2

    type = ADDiffusion
    variable = u
  [../]
  [./bf]
    type = ADBodyForce
    variable = u
    postprocessor = ramp
  [../]
[]

[Functions]
  [./ramp]
    type = ParsedFunction
    value = 't'
  [../]
[]

[Postprocessors]
  [./ramp]
    type = FunctionValuePostprocessor
    function = ramp
    execute_on = linear
  [../]
[]

    type = DirichletBC

  [../]

    type = DirichletBC

    value = 0

  dt = 1.0
