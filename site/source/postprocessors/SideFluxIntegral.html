      SideFluxIntegral

      Computes the integral of the flux over the specified boundary

      Example Input Syntax

      [Postprocessors]
  [average_temperature]
    type = ElementAverageValue
    variable = temperature
  []
  [outlet_heat_flux]
    type = ADSideFluxIntegral
    variable = temperature
    boundary = right
    diffusivity = thermal_conductivity
  []
